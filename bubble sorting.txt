//bubble sort:

#include <iostream>
using namespace std;

void Selection(int arr[], int size){
 
   int count=1;
   while(count<size){
       for(int i=0; i<size-count; i++){
           if(arr[i]>arr[i+1]){
               int temp=arr[i];
               arr[i]=arr[i+1];
               arr[i+1]=temp;
           }
       }
       count++;
   }
   
   for(int i=0; i<size; i++){
       cout << arr[i];
   }
 
   } 
  

int main() {
    int arr[5]={4,1,3,2,4};
    int size=sizeof(arr)/sizeof(int);
    
    Selection(arr, size);
    
    return 0;
}
